mainWindow.on('unmaximize', () => {
    mainWindow.setSize(240, 135);

// main.js
const { app, BrowserWindow, ipcMain } = require('electron');
const fs = require('fs').promises;
const path = require('path');
const ROOT_DIR = '/your/root/directory';
const preloadScript = `
  const { contextBridge, ipcRenderer } = require('electron');
  contextBridge.exposeInMainWorld('fileAPI', {
    readdir: (relativePath) => ipcRenderer.invoke('fs:readdir', relativePath),
    rename: (oldRelativePath, newRelativePath) => ipcRenderer.invoke('fs:rename', oldRelativePath, newRelativePath),
    copy: (srcRelativePath, destRelativePath) => ipcRenderer.invoke('fs:copy', srcRelativePath, destRelativePath),
    delete: (relativePath) => ipcRenderer.invoke('fs:delete', relativePath)
  });
`;
function getAbsolutePath(relativePath) {
  return path.join(ROOT_DIR, relativePath);
}
function createWindow() {
  const win = new BrowserWindow({
    width: 800,
    height: 600,
    webPreferences: {
      nodeIntegration: false,
      contextIsolation: true,
      preload: `data:text/javascript,${encodeURIComponent(preloadScript)}`
    }
  });
  ipcMain.handle('fs:readdir', async (_, relativePath) => {
    const absolutePath = getAbsolutePath(relativePath);
    return await fs.readdir(absolutePath);
  });
  ipcMain.handle('fs:rename', async (_, oldRelativePath, newRelativePath) => {
    const oldAbsolutePath = getAbsolutePath(oldRelativePath);
    const newAbsolutePath = getAbsolutePath(newRelativePath);
    await fs.rename(oldAbsolutePath, newAbsolutePath);
  });
  ipcMain.handle('fs:copy', async (_, srcRelativePath, destRelativePath) => {
    const srcAbsolutePath = getAbsolutePath(srcRelativePath);
    const destAbsolutePath = getAbsolutePath(destRelativePath);
    await fs.cp(srcAbsolutePath, destAbsolutePath, { recursive: true });
  });
  ipcMain.handle('fs:delete', async (_, relativePath) => {
    const absolutePath = getAbsolutePath(relativePath);
    await fs.rm(absolutePath, { recursive: true, force: true });
  });
  win.loadFile('index.html');
}
app.whenReady().then(createWindow);






## Minimal Electron Setup for Existing Browser App

### Step 1: Create Project Folder
```bash
mkdir my-app
cd my-app
```

### Step 2: Copy Your Files
```
my-app/
├── index.html      # Your main HTML
├── script.js       # Your JS scripts (or folder)
├── style.css       # Your CSS
└── ...             # Any other assets
```

### Step 3: Initialize & Install
```bash
npm init -y
npm install electron --save-dev
```

### Step 4: Create `main.js`
```javascript
const { app, BrowserWindow } = require('electron');
const path = require('path');

function createWindow() {
    const win = new BrowserWindow({
        width: 1200,
        height: 800,
        webPreferences: {
            nodeIntegration: false,
            contextIsolation: true
        }
    });
    //BrowserWindow has resizable: true and fullscreenable: true by default
    // Menu.setApplicationMenu(null) //remove menus entirely
    win.loadFile('index.html');
    win.loadFile(path.join('src', 'index.html'));
    win.webContents.openDevTools(); // Uncomment for DevTools
    //  win.webContents.on('devtools-opened', () => { win.webContents.closeDevTools(); }); //to clumsily disable devtools
    // webPreferences: { devTools: false  //This completely disables DevTools  } //add above to completely disable
}

app.whenReady().then(createWindow);
```

```json
{
```
